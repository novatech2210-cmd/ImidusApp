 CREATE FUNCTION [dbo].[GetTipTotal]
   @SalesID int,
   @PaymentTypeID tinyint
 RETURNS money
 BEGIN
   declare @TotalAmount money
   if (@PaymentTypeID in (1, 2, 6, 9, 11)) begin
     select @TotalAmount = sum(isnull(TipAmount, 0))
     from tblPayment
     where SalesID = @SalesID
       and PaymentTypeID = @PaymentTypeID
       and Voided = 0
   else if (@PaymentTypeID in (3, 4, 5, 7, 8)) begin
     select @TotalAmount = sum(isnull(TipAmount, 0))
     from tblPayment
     where SalesID = @SalesID
       and PaymentTypeID in (3, 4, 5, 7, 8)
       and Voided = 0
   else begin
     select @TotalAmount = sum(isnull(TipAmount, 0))
     from tblPayment
     where SalesID = @SalesID
       and Voided = 0
   return isnull(@TotalAmount, 0)

   @FromDateTime datetime,
   @ToDateTime datetime,
   @ByTime bit,
   @NoGroupByType bit = 0
 BEGIN
   SET NOCOUNT ON;
   declare @HoursShifted smallint
   set @FromDateTime = dbo.GetFromDateTime(@FromDateTime, @ByTime)
   set @ToDateTime = dbo.GetToDateTime(@ToDateTime, @ByTime)
   if (dbo.IsStoreClosedAfterMidnight() = 1)
     set @HoursShifted = -6

          case when @NoGroupByType = 0 then TransType else 1 /*Sale*/ end as TransType, 
          isnull(count(TransType),0) as TransCount,
